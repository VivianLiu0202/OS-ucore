57 /* *
58  * PADDR - takes a kernel virtual address (an address that points above
59  * KERNBASE),
60  * where the machine's maximum 256MB of physical memory is mapped and returns
61  * the
62  * corresponding physical address. It panics if you pass it a non-kernel
63  * virtual address.
64  * */
65 #define PADDR(kva) \
66  ({ \
67  uintptr_t __m_kva = (uintptr_t)(kva); \
68  if (__m_kva < KERNBASE) { \
69  panic("PADDR called with invalid kva %08lx", __m_kva); \
70  } \
71  __m_kva - va_pa_offset; \
72  })
73 
74 /* *
75  * KADDR - takes a physical address and returns the corresponding kernel virtual
76  * address. It panics if you pass an invalid physical address.
77  * */
78 #define KADDR(pa) \
79  ({ \
80  uintptr_t __m_pa = (pa); \
81  size_t __m_ppn = PPN(__m_pa); \
82  if (__m_ppn >= npage) { \
83  panic("KADDR called with invalid pa %08lx", __m_pa); \
84  } \
85  (void *)(__m_pa + va_pa_offset); \
86  })
